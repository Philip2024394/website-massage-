Here's how your dashboards should fetch data using document IDs:

// For TherapistDashboardPage.tsx - fetchTherapistData function
const fetchTherapistData = async (documentId: string) => {
    try {
        console.log('üìñ Fetching therapist data for document ID:', documentId);
        
        // Use Appwrite therapistService with document ID
        const therapist = await therapistService.getById(documentId);
        
        if (therapist) {
            console.log('‚úÖ Therapist data loaded:', therapist.name);
            // Update your state with therapist data
            setTherapistData(therapist);
        } else {
            console.log('‚ùå No therapist found with ID:', documentId);
        }
    } catch (error) {
        console.error('Error fetching therapist:', error);
    }
};

// For MassagePlaceAdminDashboard.tsx - similar pattern
const fetchPlaceData = async (documentId: string) => {
    try {
        console.log('üìñ Fetching place data for document ID:', documentId);
        
        // Use Appwrite placeService with document ID
        const place = await placeService.getById(documentId);
        
        if (place) {
            console.log('‚úÖ Place data loaded:', place.name);
            setPlaceData(place);
        }
    } catch (error) {
        console.error('Error fetching place:', error);
    }
};
